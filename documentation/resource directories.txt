chief/app/
	resources/
		module-archives/
			module-name.tar
			...
		modules/
			module-name/
				Processor/
					Processor.py
					...
				Butcher/
					Split.py
					Merge.py
					...
			...
		jobs/
			job-name/
				module
				data-files/
					dataset file(s)
					...
				fragment-cache/
					fragment files
					...
				results/
					result file(s)
					...
			...

		processors/
			module-name/

				...

			job-name/
